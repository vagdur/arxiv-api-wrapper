Interface ArxivQueryOptions

Options for querying the arXiv API.

Example

const options: ArxivQueryOptions = {
  search: {
    title: ['quantum computing'],
    author: ['John Doe'],
  },
  maxResults: 10,
  sortBy: 'submittedDate',
  sortOrder: 'descending',
};
